\documentclass[tikz,border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{positioning, arrows.meta, shapes, calc, backgrounds, fit, shadows}

% ========== 配色方案（CVPR 风格）==========
\definecolor{cblue}{HTML}{2196F3}
\definecolor{cgreen}{HTML}{4CAF50}
\definecolor{corange}{HTML}{FF9800}
\definecolor{cpurple}{HTML}{9C27B0}
\definecolor{cred}{HTML}{F44336}
\definecolor{ctext}{HTML}{333333}
\definecolor{cgray}{HTML}{9E9E9E}

% ========== 样式定义 ==========
\tikzset{
  % 模块样式
  module/.style={
    draw=#1!80, fill=#1!15, rounded corners=4pt,
    minimum height=0.8cm, minimum width=2cm,
    text=#1!80!black, font=\small\sffamily,
    drop shadow={shadow xshift=1pt, shadow yshift=-1pt, opacity=0.3}
  },
  module/.default=cblue,

  % 数据/输入样式
  data/.style={
    draw=#1!80, fill=#1!10, rounded corners=4pt,
    minimum height=0.7cm, minimum width=1.8cm,
    text=#1!80!black, font=\small\sffamily,
    dashed
  },

  % 处理步骤样式
  process/.style={
    draw=#1!80, fill=#1!20,
    minimum height=0.7cm, minimum width=1.6cm,
    text=#1!80!black, font=\small\sffamily
  },

  % 决策样式
  decision/.style={
    draw=#1!80, fill=#1!15,
    diamond, aspect=2,
    text=#1!80!black, font=\small\sffamily,
    inner sep=1pt
  },

  % 连线样式
  arrow/.style={->, >=Stealth, thick, draw=#1!70},
  arrow/.default=cblue,
  dashed arrow/.style={->, >=Stealth, thick, dashed, draw=cgray},

  % 分组背景样式
  group/.style={
    draw=#1!30, fill=#1!5, rounded corners=6pt,
    inner sep=12pt
  },
}

\begin{document}
\begin{tikzpicture}[>=Stealth, node distance=1.2cm]

  % ===== 标题区域 =====
  % \node[font=\large\sffamily\bfseries, text=ctext] at (0,3) {系统架构图标题};

  % ===== 示例：三层架构 =====
  % 上层
  \node[module=cblue] (app) {Application Layer};

  % 中层
  \node[module=cgreen, below=of app] (logic) {Business Logic};
  \node[module=cgreen, left=0.5cm of logic] (api) {API Gateway};
  \node[module=cgreen, right=0.5cm of logic] (auth) {Auth Service};

  % 下层
  \node[module=corange, below=of logic] (data) {Data Layer};
  \node[data=cgray, left=0.5cm of data] (cache) {Cache};
  \node[data=cgray, right=0.5cm of data] (db) {Database};

  % 连线
  \draw[arrow=cblue] (app) -- (logic);
  \draw[arrow=cblue] (api) -- (logic);
  \draw[arrow=cblue] (auth) -- (logic);
  \draw[arrow=cgreen] (logic) -- (data);
  \draw[dashed arrow] (cache) -- (data);
  \draw[dashed arrow] (db) -- (data);

  % 分组背景
  \begin{scope}[on background layer]
    \node[group=cblue, fit=(app), label={[\sffamily\small, text=cblue!80]above:Presentation}] {};
    \node[group=cgreen, fit=(api) (logic) (auth), label={[\sffamily\small, text=cgreen!80]above:Services}] {};
    \node[group=corange, fit=(cache) (data) (db), label={[\sffamily\small, text=corange!80]above:Storage}] {};
  \end{scope}

\end{tikzpicture}
\end{document}
